#include<iostream>
using namespace std;
int main() {
   int arr[3][3], r=3, c=3, i, j;
   int a[3][3] = { {1, 2, 3} , {4, 5, 6} , {7,8,9} };
   cout<<endl;
   for(i=0; i<r; i++) {
      for(j=0; j<c; j++)
      cout<<a[i][j]<<" ";
      cout<<endl;
   }
   cout<<endl;
   for(i=0; i<r; i++)
   for(j=0; j<c; j++) {
      arr[j][i] = a[i][j];
   }
   
 
   for (int i=0; i<r; i++) 
   {
       for(int j=0; j<c/2; j++)
       {
        swap(arr[i][j],arr[i][c-j-1]);
       }
   
   }
   
   for(i=0; i<c; i++) {
      for(j=0; j<r; j++)
      cout<<arr[i][j]<<" ";
      cout<<endl;
   }
   
   return 0;
}
